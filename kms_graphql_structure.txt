Examples for Query and Mutation:
================================

query queryCalls {
  allCollections {
    id
    collection_name
    kms_model_name
    fields {
      id
      name
      type
      liquor_name
    }
  }

  allEntries {
    id
    collection_name
    values
  }

  singleModelEntries(collection_name: "car_brands") {
    id
    collection_name
    values
  }

  singleEntry(collection_name: "cars", id: 18) {
    id
    collection_name
    values
  }
}

mutation mutationCalls {
  createCollection(input: {
    kms_model_name: "Sport",
    collection_name: "sports",
    description: "Sports Model"
    label_field: "name",
    fields_attributes: [
      {
        name: "Name",
        liquor_name: "name",
        type: "Kms::StringField"
      }, {
        name: "Number of teams",
        liquor_name: "number_of_teams",
        type: "Kms::StringField"
      }, {
        name: "Number of players per team",
        liquor_name: "number_of_players_per_team",
        type: "Kms::StringField"
      }, {
        type: "Kms::FileField",
        name: "Logo",
        liquor_name: "logo"
      }
    ]
  }) {
    collection {
      id
      collection_name
      kms_model_name
      fields {
        id
        liquor_name
        type
        name
        class_name
      }
    }
    errors
  }
  
  createEntry(input: {
    collection_name: "cars",
    values: [
      ["name", "BMW 760Li xDrive V12 Excellence"]
      ["brand", "3"],
      ["price", "2Cr"],
      ["user", "2"]
    ]
  }) {
    entry {
      id
      slug
      collection_name
      values
    }
    errors
  }
  
  updateEntry(input: {
    slug: "bmw-760li-xdrive-v12-excellence" # its slug of entry
    collection_name: "cars",
    values: [
      ["price", "2.1Cr"]
    ]
  }) {
    entry {
      id
      slug
      collection_name
      values
    }
    errors
  }
  
  deleteEntry(input: {
    slug: "bmw-760li-xdrive-v12-excellence" # its slug of entry
    collection_name: "cars"
  }) {
    entry {
      id
      slug
      collection_name
    }
    errors
  }
}
