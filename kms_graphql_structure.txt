Examples for Query and Mutation:
================================

query queryCalls {
  allCollections {
    id
    collection_name
    kms_model_name
  }
  allEntries {
    id
    collection_name
    values
  }
  singleModelEntries(collection_name: "car_brands") {
    id
    collection_name
    values
  }
  singleEntry(collection_name: "cars", id: 18) {
    id
    collection_name
    values
  }
}

mutation mutationCalls {
  createCollection(
    kms_model_name: "Company Type",
    collection_name: "company_types",
    description: "CompanyTypes Model"
    allow_creation_using_form: false
  ) {
    id
    collection_name
    kms_model_name
  }
  createEntry(input: {
    collection_name: "cars",
    values: [
      ["name", "BMW 760Li xDrive V12 Excellence"]
      ["brand", "3"],
      ["price", "2Cr"],
      ["user", "2"]
    ]
  }) {
    entry {
      id
      slug
      collection_name
      values
    }
    errors
  }
  updateEntry(input: {
    slug: "bmw-760li-xdrive-v12-excellence" # its slug of entry
    collection_name: "cars",
    values: [
      ["price", "2.1Cr"]
    ]
  }) {
    entry {
      id
      slug
      collection_name
      values
    }
    errors
  }
  deleteEntry(input: {
    slug: "bmw-760li-xdrive-v12-excellence" # its slug of entry
    collection_name: "cars"
  }) {
    entry {
      id
      slug
      collection_name
    }
    errors
  }
}
